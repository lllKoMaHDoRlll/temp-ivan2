CREATE TABLE users(
user_id INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(151) NOT NULL,
phone VARCHAR(20) NOT NULL UNIQUE,
email VARCHAR(40) NOT NULL UNIQUE,
password_hash varchar(32) NOT NULL
);

CREATE TABLE workers (
worker_id INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(151) NOT NULL
);

CREATE TABLE reservations (
reservation_id INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
user_id INT(10) NOT NULL FOREIGN KEY REFERENCES users(user_id),
worker_id INT(10) NOT NULL FOREIGN KEY REFERENCES workers(worker_id),
hour INT(10) NOT NULL,
date DATE NOT NULL
);
